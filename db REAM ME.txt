CREATE TABLE User(
   ID             INT PRIMARY KEY,
   UserName       varChar(50)    NOT NULL,
   PasswordHASH   text           NOT NULL,
   Presentation   int [],
   Email          varChar(50)
)

Create Table Pres(
   ID            INT FK PK,
   Presentation  text[]
)

Create Table Slides(
   Presentation  text[] fk Primary Key,
   Slides        text[]
)
